cmake_minimum_required(VERSION 3.20)
project(CustomWindowTest)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(Qt6 REQUIRED COMPONENTS Core Gui Qml Quick QuickControls2)

qt_standard_project_setup()

add_executable(custom_window_test
    test_main.cpp
)

qt_add_qml_module(custom_window_test
    URI CustomWindowTest
    VERSION 1.0
    QML_FILES
        test_custom_window.qml
)

target_link_libraries(custom_window_test PRIVATE
    Qt6::Core
    Qt6::Gui
    Qt6::Qml
    Qt6::Quick
    Qt6::QuickControls2
)
